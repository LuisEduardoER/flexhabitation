12.    private function clicou():void{

15.     arquivo.addEventListener(Event.SELECT, selecionou)
16.     /*
17.     *faz a abertura do browser para que se possa procurar o arquivo em questão e
18.     *o FileFilter serve para limitar os tipos de documentos que poderão ser aceitos na busca

20.     arquivo.browse([new FileFilter("Arquivos *.jpg", "*.jpg")])
21.    }
22. 
23.    private function selecionou(e:Event):void{
24.     //este evento é adiciona para que quando acabar de ser carregado os bytes ele exiba a imagem

26.     //função que fará a leitura do bytearray
27.     arquivo.load()
28.    }
29. 
30.    private function carregouByteArray(e:Event):void{
31.     //faz o load utilizando o byte array
